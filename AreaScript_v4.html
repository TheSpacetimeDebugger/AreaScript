💡 مثال سريع

# حاسبة تربيعية دالة احل_تربيعية(أ, ب, ج): محدد = ب**2 - 4*أ*ج اذا محدد < 0: ارجع "لا حلول حقيقية" والا_اذا محدد == 0: س = -ب / (2*أ) ارجع "حل واحد: " + نص(تقريب(س, 3)) والا: س1 = (-ب + جذر(محدد)) / (2*أ) س2 = (-ب - جذر(محدد)) / (2*أ) ارجع "حلان: " + نص(تقريب(س1,3)) + " و " + نص(تقريب(س2,3)) اطبع(احل_تربيعية(1, -5, 6))
